1  //===================================================================================
     2  // File name: x2p_core.sv
     3  // Project  : X2P
     4  // Function : IP core design of AXI to APB bridge
     5  //===================================================================================
     6  module axi_transaction_controller (// AXI protocol
     7      aclk,
     8      aresetn,
     9  	// Address write channel
     10  	awvalid,
     11  	awready,
     12  	awaddr,
     13  	awsize,
     14  	awlen,
     15  	awburst,
     16  	awid,
     17  	awprot,
     18  	// Address read channel
     19  	arvalid,
     20  	arready,
     21  	araddr,
     22  	arsize,
     23  	arlen,
     24  	arburst,
     25  	arid,
     26  	arprot,
     27  	// Write data channel
     28  	wvalid,
     29  	wready,
     30  	wdata,
     31  	wstrb,
     32  	wlast,
     33  	// Read data channel
     34  	rvalid,
     35  	rready,
     36  	rlast,
     37  	rresp,
     38  	rid,
     39  	rdata,
     40  	// Write respond channel
     41  	bvalid,
     42  	bready,
     43  	bresp,
     44  	bid,
     45  	//counter input
     46  	burst_almost_done_i,
     47  	burst_done_i,
     48  	//arbiter input 
     49  	grant_i,
     50  	//apb master input
     51  	prdata_i,
     52  	pslverr_i,
     53  	write_to_rd_sfifo_i,
     54  	read_from_wd_sfifo_i,
     55  	dec_error_i,
     56  	latch_resp_i,
     57  	//	
     58    selected_addr_o,
     59    selected_len_o,
     60    selected_size_o,
     61    selected_burst_o,
     62    selected_prot_o,
     63    	//
     64    wdata_to_apb_o,
     65    wstrb_to_apb_o,
     66    next_transfer_rdy_o,
     67    trans_go_o
     68  	  //
     69  );
     70    //iclude parameter file
     71    import parameter_pkg::*; 
     72    //ports declaration
     73    //AXI protocol
     74    input logic                     aclk;
     75    input logic                     aresetn;
     76    // Address write channel
     77    input  logic                    awvalid;
     78    input  logic [31:0]             awaddr;
     79    input  logic [2:0]              awsize;
     80    input  logic [7:0]              awlen;
     81    input  logic [1:0]              awburst;
     82    input  logic [7:0]              awid;
     83    input  logic [2:0]              awprot;
     84    output logic                    awready;
     85    // address read channel
     86    input  logic                    arvalid;
     87    input  logic [31:0]             araddr;
     88    input  logic [2:0]              arsize;
     89    input  logic [7:0]              arlen;
     90    input  logic [1:0]              arburst;
     91    input  logic [7:0]              arid;
     92    input  logic [2:0]              arprot;
     93    output logic                    arready;
     94    //write data channel
     95    input  logic                    wvalid;
     96    input  logic [31:0]             wdata;
     97    input  logic [3:0]              wstrb;
     98    input  logic                    wlast;
     99    output logic                    wready;
     100    //read data channel
     101    input  logic                    rready;
     102    output logic                    rvalid;
     103    output logic [1:0]              rresp;
     104    output logic                    rlast;
     105    output logic [7:0]              rid;
     106    output logic [31:0]             rdata;
     107    //write respond channel
     108    input  logic                    bready;
     109    output logic                    bvalid;
     110    output logic [1:0]              bresp;
     111    output logic [7:0]              bid;
     112  //counter input
     113  input logic burst_almost_done_i;
     114  input logic burst_done_i;
     115  //arbiter input 
     116  input logic [1:0] grant_i;
     117  //apb master input
     118  input logic [31:0] prdata_i;
     119  input logic pslverr_i;
     120  input logic write_to_rd_sfifo_i;
     121  input logic read_from_wd_sfifo_i;
     122  input logic dec_error_i;
     123  input logic latch_resp_i;
     124  //	
     125  output logic [31:0]  selected_addr_o;
     126  output logic [7:0]  selected_len_o;
     127  output logic [2:0]  selected_size_o;
     128  output logic [1:0]  selected_burst_o;
     129  output logic [2:0]  selected_prot_o;
     130  //
     131  output logic [31:0] wdata_to_apb_o;
     132  output logic [3:0] wstrb_to_apb_o;
     133  //
     134  output logic next_transfer_rdy_o;
     135  output logic trans_go_o;
     136    //
     137    //internal signals
     138    //
     139    //SFIFO_AW
     140    logic                           sfifo_aw_full;
     141    logic                           sfifo_aw_empty;
     142    logic                           sfifo_aw_avail;
     143    logic                           sfifo_aw_we;
     144    logic                           sfifo_aw_re;
     145    logic [31:0]                    sfifo_aw_addr;
     146    logic [7:0]                     sfifo_aw_id;
     147    logic [7:0]                     sfifo_aw_len;
     148    logic [2:0]                     sfifo_aw_size;
     149    logic [1:0]                     sfifo_aw_burst;
     150    logic [2:0]                     sfifo_aw_prot;
     151    //SFIFO_AR
     152    logic                           sfifo_ar_full;
     153    logic                           sfifo_ar_empty;
     154    logic                           sfifo_ar_avail;
     155    logic                           sfifo_ar_we;
     156    logic                           sfifo_ar_re;
     157    logic [31:0]                    sfifo_ar_addr;
     158    logic [7:0]                     sfifo_ar_id;
     159    logic [7:0]                     sfifo_ar_len;
     160    logic [2:0]                     sfifo_ar_size;
     161    logic [1:0]                     sfifo_ar_burst;
     162    logic [2:0]                     sfifo_ar_prot;
     163    //SFIFO_WD
     164    logic                           sfifo_wd_full;
     165    logic                           sfifo_wd_empty;
     166    logic                           sfifo_wd_almost_empty;
     167    logic                           sfifo_wd_we;
     168    logic                           sfifo_wd_re;
     169    logic [31:0]                    sfifo_wd_wdata;
     170    logic [3:0]                     sfifo_wd_wstrb;
     171    logic 			  sfifo_wd_last;
     172    logic	 			  wlast_flag;
     173    //SFIFO_RD
     174    logic                           sfifo_rd_almost_full;
     175    logic                           sfifo_rd_empty;
     176    logic                           sfifo_rd_we;
     177    logic                           sfifo_rd_re;
     178    //RD_CH
     179    logic[1:0]                      resp_of_rdata;
     180    logic 			  rdata_last;
     181    //AXI FSM INPUT
     182    logic                           burst_en;
     183    logic 			  new_req_rdy;
     184    //AXI FSM OUTPUT
     185    logic [3:0]			  control;
     186    //WRITE RESPONSE
     187    logic bresp_register;
     188    //
     189    parameter logic [POINTER_WIDTH:0] ALMOST_FULL_VALUE = 2**POINTER_WIDTH - 1'b1;
     190    parameter logic [POINTER_WIDTH:0] ALMOST_EMPTY_VALUE = 2**POINTER_WIDTH - 4'd14;
     191    //axi state variables
     192    axi_st axi_cs, axi_ns;
     193    //***********************************************************************************
     194    //-----------------------OUTPUT ASSIGNMENT
     195    //***********************************************************************************
     196    assign selected_addr_o = grant_i[0] ? sfifo_ar_addr : sfifo_aw_addr;
     197    assign selected_len_o = grant_i[0] ? sfifo_ar_len : sfifo_aw_len;
     198    assign selected_size_o = grant_i[0] ? sfifo_ar_size : sfifo_aw_size;
     199    assign selected_burst_o = grant_i[0] ? sfifo_ar_burst : sfifo_aw_burst;
     200    assign selected_prot_o = grant_i[0] ? sfifo_ar_prot : sfifo_aw_prot;
     201  //transaction control signal
     202  assign next_transfer_rdy_o = trans_go_o ? (grant_i[0] ? ~sfifo_rd_almost_full : ~sfifo_wd_almost_empty) : 1'b0;
     203  //state reg
     204  always_ff@(posedge aclk, negedge aresetn) begin
     205  	if(~aresetn) axi_cs <= A_IDLE;
     206  	else axi_cs <= axi_ns;
     207  end
     208  //AXI state comb circuit
     209  assign burst_en = new_req_rdy ? (grant_i[0] ? sfifo_ar_avail : sfifo_aw_avail) : 1'b0;
     210  //
     211  //ready to run new request
     212  //
     213    always_ff@(posedge aclk, negedge aresetn) begin
     214      if(~aresetn) new_req_rdy <= 1'b1;
     215      else if(control[0]) begin
     216        new_req_rdy <= 1'b0;
     217      end
     218      else if(control[1]) begin
     219        new_req_rdy <= 1'b1;
     220      end 
     221    end
     222  //
     223  //
     224  //
     225  always_comb begin
     226    axi_ns = axi_cs;
     227    trans_go_o = 1'b0;
     228    control = 4'h0;
     229    //
     230    case(axi_cs)
     231    A_IDLE: begin
     232  	  if(burst_en) begin
     233  		  axi_ns = AXI_GO;
     234  		  control = 4'h1;
     235  	  end
     236  	  else begin
     237  		  axi_ns = A_IDLE;
     238  		  control = 4'h0;
     239  	  end
     240    end
     241    AXI_GO: begin
     242  	if(burst_almost_done_i) axi_ns = AXI_DONE;
     243  	else axi_ns = AXI_GO;
     244  	//
     245  	trans_go_o = 1'b1;
     246    end
     247    AXI_DONE: begin
     248  	  if(burst_done_i) begin
     249  		  axi_ns = A_IDLE;
     250  		  control = 4'hE;
     251  	  end
     252  	  else axi_ns = AXI_DONE;
     253    end
     254    default: begin
     255  	  axi_ns = axi_cs;
     256  	  trans_go_o = 1'b0;
     257  	  control = 4'h0;
     258    end
     259    endcase
     260  end
     261  //*******************************************************************
     262  //X2P_SFIFO_AR
     263  //*******************************************************************
     264    sfiforeq #(
     265  	.DATA_WIDTH(X2P_SFIFO_AR_DATA_WIDTH),
     266         	.POINTER_WIDTH(POINTER_WIDTH)
     267    ) ar_sfifo (
     268    .clk(aclk),
     269    .rst_n(aresetn),
     270    .wr(sfifo_ar_we),
     271    .rd(sfifo_ar_re),
     272    .data_in({araddr[31:0], arid[7:0], arlen[7:0], arsize[2:0], arburst[1:0], arprot[2:0]}),
     273    .sfifo_empty(sfifo_ar_empty),
     274    .sfifo_full(sfifo_ar_full),
     275    .data_out({sfifo_ar_addr[31:0], sfifo_ar_id[7:0], sfifo_ar_len[7:0], 
     276  	  sfifo_ar_size[2:0], sfifo_ar_burst[1:0], sfifo_ar_prot[2:0]})
     277    //
     278    //
     279    );
     280    //Logic
     281    assign sfifo_ar_avail = ~sfifo_ar_empty;
     282    assign arready  	= ~sfifo_ar_full;
     283    assign sfifo_ar_we 	= arready & arvalid;
     284    assign sfifo_ar_re 	= control[2] & grant_i[0];
     285    //*******************************************************************
     286    //X2P_SFIFO_RD
     287    //*******************************************************************
     288    sfifordata #(
     289  	.DATA_WIDTH(X2P_SFIFO_RD_DATA_WIDTH), .POINTER_WIDTH(POINTER_WIDTH),
     290  	.ALMOST_FULL_VALUE(ALMOST_FULL_VALUE), .ALMOST_EMPTY_VALUE(ALMOST_EMPTY_VALUE)
     291  ) rd_sfifo(
     292    .clk(aclk),
     293    .rst_n(aresetn),
     294    .wr(sfifo_rd_we),
     295    .rd(sfifo_rd_re),
     296    .data_in({control[2], prdata_i[31:0], resp_of_rdata[1:0], sfifo_ar_id[7:0]}),
     297    .sfifo_empty(sfifo_rd_empty),
     298    .sfifo_almost_full(sfifo_rd_almost_full),
     299    .data_out({rdata_last, rdata[31:0], rresp[1:0], rid[7:0]})
     300    );
     301    //Logic
     302    assign rvalid      = ~sfifo_rd_empty;
     303    assign sfifo_rd_re = rvalid & rready;
     304    assign sfifo_rd_we = write_to_rd_sfifo_i;
     305    //RD_CH
     306    //resp_of_rdata
     307    always_comb begin
     308      if(~pslverr_i)
     309  	  resp_of_rdata = OKAY;
     310  	else if(dec_error_i)
     311  	  resp_of_rdata = DECERR;
     312  	else
     313  	  resp_of_rdata = PSLVERR;
     314    end
     315    //rlast
     316    always_comb begin
     317  	if(rvalid & rdata_last)
     318  	  rlast = 1'b1;
     319  	else
     320  	  rlast = 1'b0;	
     321    end
     322    //*******************************************************************
     323    //X2P_SFIFO_AW
     324    //*******************************************************************
     325    sfiforeq #(
     326  	  .DATA_WIDTH(X2P_SFIFO_AW_DATA_WIDTH),
     327  	  .POINTER_WIDTH(POINTER_WIDTH)
     328  ) aw_sfifo(
     329    .clk(aclk),
     330    .rst_n(aresetn),
     331    .wr(sfifo_aw_we),
     332    .rd(sfifo_aw_re),
     333    .data_in({awaddr[31:0], awid[7:0], awlen[7:0], awsize[2:0], awburst[1:0], awprot[2:0]}),
     334    .sfifo_empty(sfifo_aw_empty),
     335    .sfifo_full(sfifo_aw_full),
     336    .data_out({sfifo_aw_addr[31:0], sfifo_aw_id[7:0], sfifo_aw_len[7:0],
     337   	 sfifo_aw_size[2:0], sfifo_aw_burst[1:0], sfifo_aw_prot[2:0]})
     338    );
     339    //Logic
     340    assign sfifo_aw_avail = ~sfifo_aw_empty;
     341    assign awready 	= ~sfifo_aw_full;
     342    assign sfifo_aw_we    = awready & awvalid;
     343    assign sfifo_aw_re    = control[3] & grant_i[1];
     344    //*******************************************************************
     345    //X2P_SFIFO_WD
     346    //*******************************************************************
     347    sfifowdata #(
     348  	.DATA_WIDTH(X2P_SFIFO_WD_DATA_WIDTH), .POINTER_WIDTH(POINTER_WIDTH),
     349  	.ALMOST_FULL_VALUE(ALMOST_FULL_VALUE), .ALMOST_EMPTY_VALUE(ALMOST_EMPTY_VALUE)
     350    ) wd_sfifo(
     351    .clk(aclk),
     352    .rst_n(aresetn),
     353    .wr(sfifo_wd_we),
     354    .rd(sfifo_wd_re),
     355    .data_in({wdata[31:0], wstrb[3:0], wlast}),
     356    .sfifo_almost_empty(sfifo_wd_almost_empty),
     357    .sfifo_full(sfifo_wd_full),
     358    .data_out({wdata_to_apb_o[31:0], wstrb_to_apb_o[3:0], wlast_flag})
     359    );
     360  //logic
     361  assign wready      = ~sfifo_wd_full;
     362  assign sfifo_wd_we = wvalid & wready;
     363  assign sfifo_wd_re = grant_i[1] & read_from_wd_sfifo_i;
     364  //
     365  //B CHANNEL
     366  //
     367  //bresp register
     368  //
     369  always_ff@(posedge aclk, negedge aresetn) begin
     370    if(~aresetn) bresp_register <= 1'b0;
     371    else if(control[1]) bresp_register <= 1'b0; 
     372    else if(latch_resp_i) bresp_register <= pslverr_i;
     373  end
     374  //
     375  //INFO
     376  //
     377  always_ff @(posedge aclk, negedge aresetn) begin
     378  	if(~aresetn) begin
     379  	  bresp[1:0] <= OKAY;
     380  	  bid[7:0] <= 8'd0;
     381  	  bvalid <= 1'b0;
     382    	end
     383  	else if(control[1] & grant_i[1]) begin
     384  		  bid[7:0] <= sfifo_aw_id[7:0];
     385  		  bvalid <= 1'b1;
     386  		  //
     387  		  if(~bresp_register | ~pslverr_i)
     388  		    bresp[1:0] <= OKAY | ~{2{wlast_flag}};
     389  		  else if(dec_error_i)
     390  		    bresp[1:0] <= DECERR;
     391  		  else
     392  		    bresp[1:0] <= PSLVERR;
     393  	end
     394  	else if(bready) begin
     395  	  bresp[1:0] <= OKAY;
     396  	  bid[7:0] <= 8'd0;
     397  	  bvalid <= 1'b0;
     398  	end
     399  end
     400  //
     401  //
     402  endmodule: axi_transaction_controller
